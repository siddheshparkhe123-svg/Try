from flask_restful import Resource, reqparse
from services.pan_service import fetch_pan_details

class PANVerification(Resource):
    def post(self):
        parser = reqparse.RequestParser()
        parser.add_argument('pan', type=str, required=True, help="PAN number is required")
        parser.add_argument('consent', type=str, default='Y')
        parser.add_argument('case_id', type=str, default='123456')
        
        args = parser.parse_args()

        pan = args['pan']
        consent = args['consent']
        case_id = args['case_id']

        result = fetch_pan_details(pan, consent, case_id)
        return result, 200






from flask import Flask
from flask_restful import Api
from controllers.pan_controller import PANVerification

def init():
    app = Flask(__name__)
    api = Api(app)

    # Register the PAN verification endpoint
    api.add_resource(PANVerification, '/verify_pan')

    return app

application = init()

if __name__ == "__main__":
    application.run(debug=True)
